125  static Filter usbFilter(data, size, F_UPPERSIDEBAND);
126 
127  // take hilbert transform if necessary
128  if(type == ModulationType::USB_HILBERT
129  || type == ModulationType::LSB_HILBERT) {
130  hilbert(data, hilData, size);
131  } else if(type == ModulationType::FM_NARROW) {

135  }

142  break;

145  case ModulationType::USB_FILTERED:
146  case ModulationType::LSB_FILTERED:
